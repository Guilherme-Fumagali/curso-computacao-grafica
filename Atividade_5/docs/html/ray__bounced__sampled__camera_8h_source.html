81 void initialize() {
82 image_height = static_cast<int>(image_width / aspect_ratio);
83 image_height = (image_height < 1) ? 1 : image_height;
84
85 center = point3(0, 0, 0);
86
87 // Determine viewport dimensions.
88 auto focal_length = 1.0;
89 auto viewport_height = 2.0;
90 auto viewport_width = viewport_height * (static_cast<double>(image_width)/image_height);
91
92 // Calculate the vectors across the horizontal and down the vertical viewport edges.
93 auto viewport_u = vec3(viewport_width, 0, 0);
94 auto viewport_v = vec3(0, -viewport_height, 0);
95
96 // Calculate the horizontal and vertical delta vectors from pixel to pixel.
97 pixel_delta_u = viewport_u / image_width;
98 pixel_delta_v = viewport_v / image_height;
99
100 // Calculate the location of the upper left pixel.
101 auto viewport_upper_left =
102 center - vec3(0, 0, focal_length) - viewport_u/2 - viewport_v/2;
103 pixel00_loc = viewport_upper_left + 0.5 * (pixel_delta_u + pixel_delta_v);
104 }

115 ray get_ray(int i, int j) const {
116 // Get a randomly sampled camera ray for the pixel at location i,j.
117
118 auto pixel_center = pixel00_loc + (i * pixel_delta_u) + (j * pixel_delta_v);
119 auto pixel_sample = pixel_center + pixel_sample_square();
120
121 auto ray_origin = center;
122 auto ray_direction = pixel_sample - ray_origin;
123
124 return ray(ray_origin, ray_direction);
125 }

136 vec3 pixel_sample_square() const {
137 // Returns a random point in the square surrounding a pixel at the origin.
138 auto px = -0.5 + random_double();
139 auto py = -0.5 + random_double();
140 return (px * pixel_delta_u) + (py * pixel_delta_v);
141 }

152 color ray_color(const ray& r, int depth, const hittable& world) const {
153 hit_record rec;
154
155 // If we've exceeded the ray bounce limit, no more light is gathered.
156 if (depth <= 0)
157 return color(0,0,0);
158
159 if (world.hit(r, interval(0.001, infinity), rec)) {
160 vec3 direction = rec.normal + random_vec3_unit_vector();
161 return 0.25 * ray_color(ray(rec.p, direction), depth-1, world);
162 }
163
164 vec3 unit_direction = unit_vector(r.direction());
165 auto a = 0.5*(unit_direction.y() + 1.0);
166 return (1.0-a)*color(1.0, 1.0, 1.0) + a*color(0.5, 0.7, 1.0);
167 }
