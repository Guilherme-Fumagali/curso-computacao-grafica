112 dielectric(double index_of_refraction) : ir(index_of_refraction) {}

117 bool scatter(const ray& r_in, const hit_record& rec, color& attenuation, ray& scattered)
118 const override {
119 attenuation = color(1.0, 1.0, 1.0);
120 double refraction_ratio = rec.front_face ? (1.0/ir) : ir;
121
122 vec3 unit_direction = unit_vector(r_in.direction());
123 double cos_theta = fmin(dot(-unit_direction, rec.normal), 1.0);
124 double sin_theta = sqrt(1.0 - cos_theta*cos_theta);
125
126 bool cannot_refract = refraction_ratio * sin_theta > 1.0;
127 vec3 direction;
128
129 if (cannot_refract || reflectance(cos_theta, refraction_ratio) > random_double())
130 direction = reflect(unit_direction, rec.normal);
131 else
132 direction = refract(unit_direction, rec.normal, refraction_ratio);
133
134 scattered = ray(rec.p, direction);
135 return true;
136 }
137
138private:
139 double ir; // Index of Refraction
140
148 static double reflectance(double cosine, double ref_idx) {
149 // Use Schlick's approximation for reflectance.
150 auto r0 = (1-ref_idx) / (1+ref_idx);
151 r0 = r0*r0;
152 return r0 + (1-r0)*pow((1 - cosine),5);
153 }
