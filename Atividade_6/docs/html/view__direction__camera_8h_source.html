102 void initialize() {
103 image_height = static_cast<int>(image_width / aspect_ratio);
104 image_height = (image_height < 1) ? 1 : image_height;
105
106 center = lookfrom;
107
108 // Determine viewport dimensions.
109 auto focal_length = (lookfrom - lookat).length();
110 auto theta = degrees_to_radians(vfov);
111 auto h = tan(theta/2);
112 auto viewport_height = 2 * h * focal_length;
113 auto viewport_width = viewport_height * (static_cast<double>(image_width)/image_height);
114
115 // Calculate the u,v,w unit basis vectors for the camera coordinate frame.
116 w = unit_vector(lookfrom - lookat);
117 u = unit_vector(cross(vup, w));
118 v = cross(w, u);
119
120 // Calculate the vectors across the horizontal and down the vertical viewport edges.
121 vec3 viewport_u = viewport_width * u; // Vector across viewport horizontal edge
122 vec3 viewport_v = viewport_height * -v; // Vector down viewport vertical edge
123
124 // Calculate the horizontal and vertical delta vectors from pixel to pixel.
125 pixel_delta_u = viewport_u / image_width;
126 pixel_delta_v = viewport_v / image_height;
127
128 // Calculate the location of the upper left pixel.
129 auto viewport_upper_left = center - (focal_length * w) - viewport_u/2 - viewport_v/2;
130 pixel00_loc = viewport_upper_left + 0.5 * (pixel_delta_u + pixel_delta_v);
131 }

142 ray get_ray(int i, int j) const {
143 // Get a randomly sampled camera ray for the pixel at location i,j.
144
145 auto pixel_center = pixel00_loc + (i * pixel_delta_u) + (j * pixel_delta_v);
146 auto pixel_sample = pixel_center + pixel_sample_square();
147
148 auto ray_origin = center;
149 auto ray_direction = pixel_sample - ray_origin;
150
151 return ray(ray_origin, ray_direction);
152 }
153
163 vec3 pixel_sample_square() const {
164 // Returns a random point in the square surrounding a pixel at the origin.
165 auto px = -0.5 + random_double();
166 auto py = -0.5 + random_double();
167 return (px * pixel_delta_u) + (py * pixel_delta_v);
168 }

179 color ray_color(const ray& r, int depth, const hittable& world) const {
180 // If we've exceeded the ray bounce limit, no more light is gathered.
181 if (depth <= 0)
182 return color(0,0,0);
183
184 hit_record rec;
185 if (world.hit(r, interval(0.001, infinity), rec)) {
186 ray scattered;
187 color attenuation;
188 if (rec.mat->scatter(r, rec, attenuation, scattered))
189 return attenuation * ray_color(scattered, depth-1, world);
190 return color(0,0,0);
191 }
192
193 vec3 unit_direction = unit_vector(r.direction());
194 auto a = 0.5*(unit_direction.y() + 1.0);
195 return (1.0-a)*color(1.0, 1.0, 1.0) + a*color(0.5, 0.7, 1.0);
196 }
